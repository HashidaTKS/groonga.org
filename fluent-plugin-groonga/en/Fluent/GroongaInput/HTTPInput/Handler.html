def on_message_complete
  params = WEBrick::HTTPUtils.parse_query(@parser.query_string)
  path_info = @parser.request_path
  case path_info
  when /\A\/d\//
    command = $POSTMATCH
    if command == "load"
      params["values"] = @body unless @body.empty?
    end
    @input.emit(command, params)
  end
end
