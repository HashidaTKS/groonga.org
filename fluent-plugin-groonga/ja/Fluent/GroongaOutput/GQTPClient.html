def send(command)
  @client ||= GQTP::Client.new(:address => @host,
                               :port => @port,
                               :connection => :coolio,
                               :loop => @loop)
  @client.send(command.to_command_format) do |header, body|
    @loop.stop
  end
  @loop.run
end

def shutdown
  return if @client.nil?
  @client.close do
    @loop.stop
  end
  @loop.run
end
