def send(command)
  body = nil
  if command.name == "load"
    body = command.arguments.delete(:values)
  end
  @groonga_input.write("#{command.to_uri_format}\n")
  if body
    body.each_line do |line|
      @groonga_input.write("#{line}\n")
    end
  end
  @loop.run
end
