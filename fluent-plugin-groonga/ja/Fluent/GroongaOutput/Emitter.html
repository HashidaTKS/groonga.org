def emit(chunk)
  records = []
  chunk.msgpack_each do |message|
    tag, _, record = message
    if /\Agroonga\.command\./ =~ tag
      name = $POSTMATCH
      unless records.empty?
        store_records(records)
        records.clear
      end
      @client.execute(name, record)
    else
      records << record
    end
  end
  store_records(records) unless records.empty?
end
