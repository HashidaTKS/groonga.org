def run(*command_line)
  option_parser = create_option_parser
  begin
    option_parser.parse!(command_line)
  rescue OptionParser::ParseError => error
    $stderr.puts(error.message)
    return false
  end

  tester = Tester.new(old_groonga_server,
                      new_groonga_server,
                      tester_options)
  tester.run
end
