def initialize(old, new, options)
  @old = old
  @new = new
  @input_directory = options[:input_directory] || Pathname.new(".")
  @working_directory = options[:working_directory] || Pathname.new(".")

  @options = options
  @n_ready_waits = 2
  @clone_pids = []
end

def run
  old_thread = Thread.new do
    @old.run do
      run_test
    end
  end
  new_thread = Thread.new do
    @new.run do
      run_test
    end
  end

  old_thread.join
  new_thread.join
end
