def report_statistic(statistic)
  @index += 1
  write("%*d) %s" % [@digit, @index, format_heading(statistic)])
  report_parameters(statistic)
  report_operations(statistic)
end

def start
  @index = 0
  if @statistics.size.zero?
    @digit = 1
  else
    @digit = Math.log10(@statistics.size).truncate + 1
  end
end
