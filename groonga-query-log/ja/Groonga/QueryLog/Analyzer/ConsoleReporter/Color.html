def sequence
  sequence = []
  if @name == "none"
  elsif @name == "reset"
    sequence << "0"
  else
    foreground_parameter = foreground? ? 3 : 4
    foreground_parameter += 6 if intensity?
    sequence << "#{foreground_parameter}#{NAMES.index(@name)}"
  end
  sequence << "1" if bold?
  sequence << "3" if italic?
  sequence << "4" if underline?
  sequence
end
